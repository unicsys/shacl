@prefix hamr: <http://hamr.org/ontology#> .
@prefix time: <http://www.w3.org/2006/time#> .
@prefix owl: <http://www.w3.org/2002/07/owl#> .
@prefix rdfs: <http://www.w3.org/2000/01/rdf-schema#> .
@prefix xsd: <http://www.w3.org/2001/XMLSchema#> .

# ============ Core Classes ============

hamr:Engine a owl:Class ;
    rdfs:label "Aircraft Engine" ;
    rdfs:comment "An aircraft engine with trackable configuration" .

hamr:Part a owl:Class ;
    rdfs:label "Engine Part" ;
    rdfs:comment "A component that can be installed on an engine" .

hamr:PartInstallation a owl:Class ;
    rdfs:label "Part Installation Event" ;
    rdfs:comment "A temporal event representing installation of a part on an engine" .

hamr:PartRemoval a owl:Class ;
    rdfs:label "Part Removal Event" ;
    rdfs:comment "A temporal event representing removal of a part from an engine" .

hamr:EngineLocation a owl:Class ;
    rdfs:label "Engine Location" ;
    rdfs:comment "A specific location/position on an engine where parts can be installed" .

hamr:PartConfiguration a owl:Class ;
    rdfs:label "Part Configuration State" ;
    rdfs:comment "The complete configuration of parts on an engine at a point in time" .

# ============ Object Properties ============

# Core relationships
hamr:hasInstalledPart a owl:ObjectProperty ;
    rdfs:domain hamr:Engine ;
    rdfs:range hamr:Part ;
    rdfs:comment "Currently installed part (current state only)" .

hamr:installedOn a owl:ObjectProperty ;
    rdfs:domain hamr:Part ;
    rdfs:range hamr:Engine ;
    owl:inverseOf hamr:hasInstalledPart .

hamr:atLocation a owl:ObjectProperty ;
    rdfs:domain hamr:PartInstallation ;
    rdfs:range hamr:EngineLocation ;
    rdfs:comment "Specific location where part is installed" .

# Temporal relationships
hamr:hasInstallationEvent a owl:ObjectProperty ;
    rdfs:domain hamr:Part ;
    rdfs:range hamr:PartInstallation .

hamr:hasRemovalEvent a owl:ObjectProperty ;
    rdfs:domain hamr:Part ;
    rdfs:range hamr:PartRemoval .

hamr:installationOfPart a owl:ObjectProperty ;
    rdfs:domain hamr:PartInstallation ;
    rdfs:range hamr:Part ;
    owl:inverseOf hamr:hasInstallationEvent .

hamr:removalOfPart a owl:ObjectProperty ;
    rdfs:domain hamr:PartRemoval ;
    rdfs:range hamr:Part ;
    owl:inverseOf hamr:hasRemovalEvent .

hamr:installedOnEngine a owl:ObjectProperty ;
    rdfs:domain hamr:PartInstallation ;
    rdfs:range hamr:Engine .

hamr:removedFromEngine a owl:ObjectProperty ;
    rdfs:domain hamr:PartRemoval ;
    rdfs:range hamr:Engine .

# Work order relationships
hamr:performedDuringWorkOrder a owl:ObjectProperty ;
    rdfs:domain hamr:PartInstallation ;
    rdfs:range hamr:WorkOrder .

hamr:resultedInRemoval a owl:ObjectProperty ;
    rdfs:domain hamr:WorkOrder ;
    rdfs:range hamr:PartRemoval .

hamr:resultedInInstallation a owl:ObjectProperty ;
    rdfs:domain hamr:WorkOrder ;
    rdfs:range hamr:PartInstallation .

# Part properties

    rdfs:domain hamr:Part ;
    rdfs:range xsd:string ;
    rdfs:comment "Part number (PN)" .

hamr:serialNumber a owl:DatatypeProperty ;
    rdfs:domain hamr:Part ;
    rdfs:range xsd:string ;
    rdfs:comment "Part serial number (SN)" .

hamr:partNomenclature a owl:DatatypeProperty ;
    rdfs:domain hamr:Part ;
    rdfs:range xsd:string .

# Engine properties
hamr:engineSerialNumber a owl:DatatypeProperty ;
    rdfs:domain hamr:Engine ;
    rdfs:range xsd:string .

hamr:engineId a owl:DatatypeProperty ;
    rdfs:domain hamr:Engine ;
    rdfs:range xsd:string .

# Location properties
hamr:locationCode a owl:DatatypeProperty ;
    rdfs:domain hamr:EngineLocation ;
    rdfs:range xsd:string ;
    rdfs:comment "LCN code" .

hamr:locationDescription a owl:DatatypeProperty ;
    rdfs:domain hamr:EngineLocation ;
    rdfs:range xsd:string .

# Temporal properties
hamr:installationDate a owl:DatatypeProperty ;
    rdfs:domain hamr:PartInstallation ;
    rdfs:range xsd:dateTime .

hamr:removalDate a owl:DatatypeProperty ;
    rdfs:domain hamr:PartRemoval ;
    rdfs:range xsd:dateTime .

hamr:recordedDate a owl:DatatypeProperty ;
    rdfs:comment "Date when the record was created in the system" ;
    rdfs:range xsd:dateTime .

# ============ Temporal Interval Class ============

hamr:PartInstallationInterval a owl:Class ;
    rdfs:label "Part Installation Interval" ;
    rdfs:comment "Time interval during which a part was installed" .

hamr:hasInterval a owl:ObjectProperty ;
    rdfs:domain hamr:Part ;
    rdfs:range hamr:PartInstallationInterval .

hamr:intervalStart a owl:DatatypeProperty ;
    rdfs:domain hamr:PartInstallationInterval ;
    rdfs:range xsd:dateTime .

hamr:intervalEnd a owl:DatatypeProperty ;
    rdfs:domain hamr:PartInstallationInterval ;
    rdfs:range xsd:dateTime .

hamr:duringInterval a owl:ObjectProperty ;
    rdfs:domain hamr:PartInstallationInterval ;
    rdfs:range hamr:Engine .

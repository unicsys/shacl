PREFIX : <http://example.org/yourdid#>
PREFIX dc: <http://purl.org/dc/elements/1.1/>
PREFIX bfo: <http://purl.obolibrary.org/obo/BFO_>

PREFIX rdf: <http://www.w3.org/1999/02/22-rdf-syntax-ns#>
SELECT ?title ?identifier ?approvalDateValue
WHERE {
  # Get the title and identifier directly from DI-SESS-81000F
  :DI-SESS-81000F dc:title ?title ;
                  dc:identifier ?identifier .

  # Navigate to the approval date
  # Path: DI-SESS-81000F -> isOutputOf -> ApprovalProcess -> occupies_temporal_region -> TemporalRegion -> has_last_instant -> Instant -> dc:date
  OPTIONAL { # Using OPTIONAL in case some part of the approval date path is missing for some data
    :DI-SESS-81000F :isOutputOf ?approvalProcess .
    ?approvalProcess rdf:type :ApprovalProcess ; # Ensure it's an approval process
                     bfo:0000199 ?temporalRegion . # bfo:occupies_temporal_region
    ?temporalRegion bfo:0000224 ?approvalInstant . # bfo:has_last_instant
    ?approvalInstant dc:date ?approvalDateValue .
  }
}
